CREATE TRIGGER TRIG1 ON EMPLOYEE
FOR UPDATE, INSERT
AS
IF (SELECT EMP_DOB FROM INSERTED) < '01/01/50'  
  BEGIN
  PRINT 'Incorrect DOB Entry. Please Enter A Later DOB.'
  ROLLBACK TRANSACTION
  END

IF (SELECT EMP_DOB FROM INSERTED) > '01/01/90'
  BEGIN
  PRINT 'Incorrect DOB Entry. Please Enter An Earlier DOB.'
  ROLLBACK
  END
