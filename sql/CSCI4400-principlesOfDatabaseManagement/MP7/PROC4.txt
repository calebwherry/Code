CREATE PROCEDURE PROC4 @EMP_CODE INT
AS

  IF NOT EXISTS (SELECT * FROM EMPLOYEE E, STORE S WHERE E.EMP_CODE = S.EMP_CODE AND S.EMP_CODE = @EMP_CODE)
    PRINT 'This is not a manager!'
  ELSE
    BEGIN
      SELECT S.STORE_CODE, STORE_NAME, STORE_YTD_SALES, REGION_CODE
      FROM STORE S, EMPLOYEE E
      WHERE S.EMP_CODE = E.EMP_CODE AND S.EMP_CODE = @EMP_CODE
    END
